CREACIÓN DE TABLAS


HOSPITAL:

CREATE TABLE HOSPITAL(
ID_HOSPITAL NUMBER(5),
NOMBRE_HOSPITAL VARCHAR2(50),
DIRECCION_HOSPITAL VARCHAR2(50),
TELEFONO_HOSPITAL NUMBER(9),
CONSTRAINT PK_ID_HOSPITAL PRIMARY KEY (ID_HOSPITAL),
CONSTRAINT NN_NOMBRE_HOSPITAL CHECK (NOMBRE_HOSPITAL IS NOT NULL),
CONSTRAINT NN_DIRECCION_HOSPITAL CHECK (DIRECCION_HOSPITAL IS NOT NULL),
CONSTRAINT NN_TELEFONO_HOSPITAL CHECK (TELEFONO_HOSPITAL IS NOT NULL)
);


LABORATORIO:

CREATE TABLE LABORATORIO(
ID_LABORATORIO NUMBER(5),
ID_HOSPITAL NUMBER(5),
NOMBRE_LABORATORIO VARCHAR2(50),
DIRECCION_LABORATORIO VARCHAR2(50),
TELEFONO_LABORATORIO NUMBER(9),
CONSTRAINT PK_ID_LABORATORIO PRIMARY KEY (ID_LABORATORIO),
CONSTRAINT FK_ID_HOSPITAL FOREIGN KEY (ID_HOSPITAL) REFERENCES HOSPITAL (ID_HOSPITAL),
CONSTRAINT NN_NOMBRE_LABORATORIO CHECK (NOMBRE_LABORATORIO IS NOT NULL),
CONSTRAINT NN_DIRECCION_LABORATORIO CHECK (DIRECCION_LABORATORIO IS NOT NULL),
CONSTRAINT NN_TELEFONO_LABORATORIO CHECK (TELEFONO_LABORATORIO IS NOT NULL)
);


SALAS:

CREATE TABLE SALAS(
ID_SALA NUMBER(5),
ID_HOSPITAL NUMBER(5),
NOMBRE_SALA VARCHAR2(50),
CANTIDAD_CAMAS_SALA NUMBER(2),
CONSTRAINT PK_ID_SALA PRIMARY KEY (ID_SALA),
CONSTRAINT FKK_ID_HOSPITAL FOREIGN KEY (ID_HOSPITAL) REFERENCES HOSPITAL (ID_HOSPITAL),
CONSTRAINT NN_ID_SALA CHECK (ID_SALA IS NOT NULL),
CONSTRAINT NN_NOMBRE_SALA CHECK (NOMBRE_SALA IS NOT NULL),
CONSTRAINT NN_CANTIDAD_CAMAS_SALA CHECK (CANTIDAD_CAMAS_SALA IS NOT NULL)
);


PACIENTE:

CREATE TABLE PACIENTE(
ID_PACIENTE NUMBER(5),
NOMBRE_PACIENTE VARCHAR2(30),
SEXO_PACIENTE VARCHAR2(20),
FECHA_NAC_PACIENTE DATE,
NUM_REGISTRO_PACIENTE NUMBER(5),
NUM_CAMA_PACIENTE NUMBER(5),
DIRECCION_PACIENTE VARCHAR2(40),
CONSTRAINT PK_ID_PACIENTE PRIMARY KEY (ID_PACIENTE),
CONSTRAINT NN_NOMBRE_PACIENTE CHECK (NOMBRE_PACIENTE IS NOT NULL),
CONSTRAINT NN_SEXO_PACIENTE CHECK (SEXO_PACIENTE IS NOT NULL),
CONSTRAINT NN_FECHA_NAC_PACIENTE CHECK (FECHA_NAC_PACIENTE IS NOT NULL),
CONSTRAINT NN_NUM_REGISTRO_PACIENTE CHECK (NUM_REGISTRO_PACIENTE IS NOT NULL),
CONSTRAINT NN_NUM_CAMA_PACIENTE CHECK (NUM_CAMA_PACIENTE IS NOT NULL),
CONSTRAINT NN_DIRECCION_PACIENTE CHECK (DIRECCION_PACIENTE IS NOT NULL),
CONSTRAINT UQ_NUM_REGISTRO_PACIENTE UNIQUE (NUM_REGISTRO_PACIENTE)
);


SALAS_PACIENTE  -> INGRESAR / INGRESO: 

CREATE TABLE SALAS_PACIENTE(
ID_SALA NUMBER(5),
ID_PACIENTE NUMBER(5),
FECHA_INGRESO DATE,
FECHA_SALIDA DATE,
CONSTRAINT FK_ID_SALA FOREIGN KEY (ID_SALA) REFERENCES SALAS (ID_SALA),
CONSTRAINT FK_ID_PACIENTE FOREIGN KEY (ID_PACIENTE) REFERENCES PACIENTE (ID_PACIENTE),
CONSTRAINT NN_FECHA_INGRESO CHECK (FECHA_INGRESO IS NOT NULL)
);


DIAGNOSTICO:

CREATE TABLE DIAGNOSTICO(
ID_DIAGNOSTICO NUMBER(5),
ID_PACIENTE NUMBER(5),
TIPO_DIAGNOSTICO VARCHAR2(30),
COMPLICACIONES_DIAGNOSTICO VARCHAR2(30),
CONSTRAINT PK_ID_DIAGNOSTICO PRIMARY KEY (ID_DIAGNOSTICO),
CONSTRAINT FKK_ID_PACIENTE FOREIGN KEY (ID_PACIENTE) REFERENCES PACIENTE (ID_PACIENTE),
CONSTRAINT NN_TIPO_DIAGNOSTICO CHECK (TIPO_DIAGNOSTICO IS NOT NULL)
);
