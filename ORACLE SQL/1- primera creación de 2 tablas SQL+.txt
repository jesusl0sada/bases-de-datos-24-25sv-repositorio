DOS TABLAS ALUMNO - CLASE : 1 ALUMNO PERTENECE A 1 CLASE, PERO A ESA CLASE PERTENECEN VARIOS ALUMNOS ;  POR ENDE N ALUMNOS PERTENCEN A 1 CLASE; 


RELACIÓN ALUMNOS-CLASE: N-1. 

CREATE TABLE CLASE(
id_clase VARCHAR2 (3), 
nombre_clase VARCHAR2 (15),
lugar_clase VARCHAR2 (10),
CONSTRAINT PK_ID_CLASE primary key (id_clase),
CONSTRAINT NN_nombre_clase CHECK (nombre_clase IS NOT NULL),
CONSTRAINT NN_lugar_clase CHECK (lugar_clase IS NOT NULL)
); 


CREATE TABLE ALUMNO(
num_mat NUMBER(5),
nombre VARCHAR2(20),
apellido VARCHAR2(40),
direccion VARCHAR2(50),
fecha_nac DATE,
id_clase VARCHAR2 (3),
CONSTRAINT CK_FECHA_NAC CHECK (fecha_nac < TO_DATE('13/01/2013', 'DD-MM-YYYY')),
CONSTRAINT PK_ALUMNO PRIMARY KEY (num_mat),
CONSTRAINT NN_NOMBRE_ALUMNO CHECK (nombre IS NOT NULL),
CONSTRAINT FK_ID_CLASE FOREIGN KEY (id_clase) REFERENCES CLASE (id_clase)
);

// SI HAY UNA RELACIÓN ES CON REFERENCES, YA QUE ES UN DESPLEGABLE DE LOS DATOS DE LA OTRA TABLA Y LO SETEAMOS COMO FOREIGN KEY PARA QUE NO SE REPITA LA COMBINACIÓN
