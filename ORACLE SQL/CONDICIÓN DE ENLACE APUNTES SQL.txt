-- LOS EMPLEADOS QUE TRABAJAN EN BARCELONA 

SELECT EMP_NO, APELLIDO, OFICIO (AQUÍ HAY QUE ESPECIFICAR YA QUE VAS A ELEGIR UNA PARTE DE UNA TABLA PERO NECESITAS QUE TENGA MEMORIA DE LAS 2 POR LO TANTO EN FROM ESTÁN AMBAS)
FROM EMPLE, DEPART (HAY QUE SELECCIONAR EN AMBAS SI LA CONDICION Y ELEGIR VA A ESTAR EN LOS DOS)
// CONDICIONES WHERE EMPLE.DEPT_NO = DEPART.DEPT_NO AND LOC = 'BARCELONA'; (EN LA RELACIÓN 1-N ESTE DATO ES EL MISMO EN AMBOS, SE PUEDE HACER TAMBIÉN CON EL JOIN) 

CÓDIGO SIN COMENTAR:
SELECT EMP_NO, APELLIDO , OFICIO
FROM EMPLE, DEPART
WHERE EMPLE.DEPT_NO = DEPART.DEPT_NO AND LOC = 'BARCELONA';

MÁS COMPLEJO: WHERE EMPLE.DEPT_NO = DEPART.DEPT_NO --> ES LA CONDICIÓN DE ENLACE, LO MÁS IMPORTANTE, TE ESTÁ DICIENDO QUE EN UNA RELACIÓN 1-N
AMBAS TABLAS ESTÁN UNIDAS POR EL DEPT_NO, ES DECIR HAY QUE PONERLE ESA CONDICIÓN SIEMPRE CON LO QUE TIENEN EN COMÚN. 

N TABLAS = N-1 CONDICIONES DE ENLACE

2 TABLAS ---> 1 CONDICIÓN
3 TABLAS ---> 2 CONDICIONES
4 TABLAS ---> 3 CONDICIONES
5 TABLAS ---> 4 CONDICIONES
